 41class EvaluationPipeline(CorePipeline):
 42    """Evaluation Pipeline to run metric computations for ratings related to a dataset.
 43
 44    The pipeline is intended to be used multiple times on different computed rating files
 45    that are all associated to a specific dataset.
 46    Loading the evaluation sets is done each time for every metric configuration, so that
 47    they can be filtered individually on subgroups before computing the actual evaluation.
 48    For each metric it executes the following steps:
 49
 50    1) create the metric.
 51    2) load the evaluation sets.
 52    3) filter the evaluation sets (optional).
 53    4) compute the evaluation of the metric.
 54    5) store evaluations in an overview file.
 55
 56    Public methods:
 57
 58    run
 59    """
 60
 61    def __init__(
 62            self,
 63            dataset: Dataset,
 64            data_filter_factory: GroupFactory,
 65            metric_category_factory: GroupFactory,
 66            event_dispatcher: EventDispatcher):
 67        """Construct the evaluation pipeline.
 68
 69        Args:
 70            dataset: the dataset that is associated with the evaluation of the rating sets.
 71            data_filter_factory: the factory with available filters for all dataset-matrix pairs.
 72            metric_category_factory: the metric category factory with available metric factories.
 73            event_dispatcher: used to dispatch model/IO events when running the pipeline.
 74        """
 75        CorePipeline.__init__(self, event_dispatcher)
 76        self.dataset = dataset
 77        self.data_filter_factory = data_filter_factory
 78        self.metric_category_factory = metric_category_factory
 79
 80    def run(self,
 81            output_path: str,
 82            eval_set_paths: EvaluationSetPaths,
 83            metrics: List[MetricConfig],
 84            is_running: Callable[[], bool]) -> None:
 85        """Run the entire pipeline from beginning to end.
 86
 87        Effectively running all computations of the specified metrics.
 88        All the specified metric configurations that have a subgroup are expected
 89        to be related to the dataset that was used to construct the pipeline.
 90
 91        Args:
 92            output_path: the path of the json file to store the output.
 93            eval_set_paths: the file paths of the evaluation sets.
 94            metrics: list of MetricConfig objects to compute.
 95            is_running: function that returns whether the pipeline
 96                is still running. Stops early when False is returned.
 97        """
 98        self.event_dispatcher.dispatch(EvaluationPipelineEventArgs(
 99            ON_BEGIN_EVAL_PIPELINE,
100            metrics
101        ))
102
103        start = time.time()
104
105        # Create evaluations file
106        create_json(
107            output_path,
108            {'evaluations': []},
109            self.event_dispatcher,
110            indent=4
111        )
112
113        for metric_config in metrics:
114            if not is_running():
115                return
116
117            metric_factory = resolve_factory(
118                metric_config.name,
119                self.metric_category_factory
120            )
121            if metric_factory is None:
122                self.event_dispatcher.dispatch(ErrorEventArgs(
123                    ON_FAILURE_ERROR,
124                    'Failure: to resolve metric factory for metric ' + metric_config.name
125                ))
126                continue
127
128            try:
129                self.run_metric(output_path, metric_factory, eval_set_paths, metric_config)
130            except ArithmeticError:
131                self.event_dispatcher.dispatch(ErrorEventArgs(
132                    ON_RAISE_ERROR,
133                    'ArithmeticError: trying to run metric ' + metric_config.name
134                ))
135                continue
136            except FileNotFoundError:
137                self.event_dispatcher.dispatch(ErrorEventArgs(
138                    ON_RAISE_ERROR,
139                    'FileNotFoundError: trying to run metric ' + metric_config.name
140                ))
141                continue
142            except MemoryError:
143                self.event_dispatcher.dispatch(ErrorEventArgs(
144                    ON_RAISE_ERROR,
145                    'MemoryError: trying to run metric ' + metric_config.name
146                ))
147                continue
148            except RuntimeError:
149                self.event_dispatcher.dispatch(ErrorEventArgs(
150                    ON_RAISE_ERROR,
151                    'RuntimeError: trying to run metric ' + metric_config.name
152                ))
153                continue
154
155        end = time.time()
156
157        self.event_dispatcher.dispatch(EvaluationPipelineEventArgs(
158            ON_END_EVAL_PIPELINE,
159            metrics
160        ), elapsed_time=end - start)
161
162    def run_metric(
163            self,
164            output_path: str,
165            metric_factory: Factory,
166            eval_set_paths: EvaluationSetPaths,
167            metric_config: MetricConfig) -> None:
168        """Run the evaluation computation for the specified metric configuration.
169
170        Args:
171            output_path: the path of the json file to store the output.
172            metric_factory: the factory that contains the specified metric.
173            eval_set_paths: the file paths of the evaluation sets.
174            metric_config: the metric evaluation configuration.
175
176        Raises:
177            ArithmeticError: possibly raised by a metric on construction or evaluation.
178            MemoryError: possibly raised by a metric on construction or evaluation.
179            RuntimeError: possibly raised by a metric on construction or evaluation.
180            FileNotFoundError: when the file of one of the evaluation sets does not exist.
181        """
182        self.event_dispatcher.dispatch(MetricEventArgs(
183            ON_BEGIN_METRIC,
184            metric_config
185        ))
186
187        start = time.time()
188
189        metric = metric_factory.create(
190            metric_config.name,
191            metric_config.params
192        )
193
194        # this can raise a FileNotFoundError, effectively aborting the pipeline
195        eval_sets = self.load_evaluation_sets(
196            eval_set_paths,
197            metric.requires_train_set,
198            metric.requires_test_set
199        )
200
201        eval_sets = self.filter_set_rows(
202            eval_sets,
203            metric_config.subgroup
204        )
205
206        evaluation = self.compute_metric_evaluation(
207            metric,
208            eval_sets,
209            metric_config
210        )
211
212        add_evaluation_to_file(output_path, evaluation, metric_config)
213
214        end = time.time()
215
216        self.event_dispatcher.dispatch(MetricEventArgs(
217            ON_END_METRIC,
218            metric_config
219        ), elapsed_time=end - start)

221    def load_evaluation_sets(
222            self,
223            eval_set_paths: EvaluationSetPaths,
224            train_set_required: bool,
225            test_set_required: bool) -> EvaluationSets:
226        """Load the required evaluation sets.
227
228        Args:
229            eval_set_paths: the file paths of the evaluation sets.
230            train_set_required: whether the train set is required for the evaluation.
231            test_set_required: whether the test set is required for the evaluation.
232
233        Raises:
234            FileNotFoundError: when the file of one of the evaluation sets does not exist.
235
236        Returns:
237            the loaded evaluation sets.
238        """
239        train_set = None if not train_set_required else self.read_dataframe(
240            eval_set_paths.train_path,
241            'train set',
242            ON_BEGIN_LOAD_TRAIN_SET,
243            ON_END_LOAD_TRAIN_SET,
244            names=['user', 'item', 'rating']
245        )
246
247        test_set = None if not test_set_required else self.read_dataframe(
248            eval_set_paths.test_path,
249            'test set',
250            ON_BEGIN_LOAD_TEST_SET,
251            ON_END_LOAD_TEST_SET,
252            names=['user', 'item', 'rating']
253        )
254
255        rating_set = self.read_dataframe(
256            eval_set_paths.ratings_path,
257            'rating set',
258            ON_BEGIN_LOAD_RATING_SET,
259            ON_END_LOAD_RATING_SET,
260        )
261
262        return EvaluationSets(train_set, test_set, rating_set)

264    def filter_set_rows(
265            self,
266            eval_sets: EvaluationSets,
267            subgroup: Optional[DataSubsetConfig]) -> EvaluationSets:
268        """Filter the evaluation set rows for the specified subgroup.
269
270        The subset is created by applying multiple filter passes to the evaluation sets
271        individually. These filter passes are then combined to form the resulting sets.
272
273        Args:
274            eval_sets: the evaluation sets to filter.
275            subgroup: the subgroup to create of the evaluation sets.
276
277        Returns:
278            the filtered evaluation sets.
279        """
280        # early exit, because no filtering is needed
281        if subgroup is None or len(subgroup.filter_passes) == 0:
282            return eval_sets
283
284        self.event_dispatcher.dispatch(FilterDataframeEventArgs(
285            ON_BEGIN_FILTER_RECS,
286            subgroup
287        ))
288
289        start = time.time()
290        # TODO filter sets using the given filters and dataset
291        end = time.time()
292
293        self.event_dispatcher.dispatch(FilterDataframeEventArgs(
294            ON_END_FILTER_RECS,
295            subgroup
296        ), elapsed_time=end - start)
297
298        return eval_sets

307        Args:
308            metric: the metric to use for computing the evaluation.
309            eval_sets: the evaluation sets to compute the performance of.
310            metric_config: the metric configuration that is associated with the metric.

314            MemoryError: possibly raised by a metric on evaluation.

321            ON_BEGIN_EVAL_METRIC,
322            metric_config

325        start = time.time()
326
327        evaluation = metric.evaluate(eval_sets)
328
329        end = time.time()
330
331        self.event_dispatcher.dispatch(MetricEventArgs(
332            ON_END_EVAL_METRIC,
333            metric_config
334        ), elapsed_time=end - start)

339def add_evaluation_to_file(
340        file_path: str,
341        evaluation_value: float,
342        metric_config: MetricConfig) -> None:
343    """Add an evaluation result to the list in the overview file.
344
345    Args:
346        file_path: the path to the evaluations overview file.
347        evaluation_value: the evaluation result.
348        metric_config: the metric configuration used for the evaluation.
349    """
350    subgroup = {} if metric_config.subgroup is None else metric_config.subgroup.to_yml_format()
351    evaluation = {'name': metric_config.name,
352                  'params': metric_config.params,
353                  'evaluation': {'value': str(evaluation_value), 'subgroup': subgroup}}
354
355    evaluations = load_json(file_path)
356    evaluations['evaluations'].append(evaluation)
357    save_json(file_path, evaluations, indent=4)
